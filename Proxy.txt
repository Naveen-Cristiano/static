#!/bin/bash

# Start Nginx
systemctl start nginx

# Start auth_service
cd /home/ec2-user/microservices/auth_service
nohup gunicorn -w 3 -b 127.0.0.1:5001 app:app &

# Start player_service
cd /home/ec2-user/microservices/player_service
nohup gunicorn -w 3 -b 127.0.0.1:5002 app:app &

# Start frontend
cd /home/ec2-user/microservices/frontend
nohup gunicorn -w 3 -b 127.0.0.1:5000 app:app &
